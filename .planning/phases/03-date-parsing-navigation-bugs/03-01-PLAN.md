```typescript
function parseDateFromFilename(filename: string): Date | null {
  // Remove file extension
  const nameWithoutExt = filename.replace(/\.[^/.]+$/, "");

  // Try common daily note formats
  const formats = [
    /(\d{4}-\d{2}-\d{2})/,    // YYYY-MM-DD            line 425
    /(\d{4}_\d{2}_\d{2})/,    // YYYY_MM_DD            line 427
    /(\d{4}\.\d{2}\.\d{2})/,  // YYYY.MM.DD            line 429
    /(\d{2}-\d{2}-\d{4})/,    // DD-MM-YYYY            line 431
    /(\d{2}-\d{2}-\d{4})/,    // MM-DD-YYYY (DUP)      line 433
    /(\d{8})/,                // YYYYMMDD              line 435
  ];

  for (const format of formats) {
    const match = nameWithoutExt.match(format);
    if (match) {
      const dateStr = match[1];

      if (/^\d{4}-\d{2}-\d{2}$/.test(dateStr)) {
        const date = new Date(dateStr);                              // BUG-01 line 446
        if (!isNaN(date.getTime())) return date;
      } else if (/^\d{4}_\d{2}_\d{2}$/.test(dateStr)) {
        const date = new Date(dateStr.replace(/_/g, "-"));           // BUG-01 line 450
        if (!isNaN(date.getTime())) return date;
      } else if (/^\d{4}\.\d{2}\.\d{2}$/.test(dateStr)) {
        const date = new Date(dateStr.replace(/\./g, "-"));          // BUG-01 line 454
        if (!isNaN(date.getTime())) return date;
      } else if (/^\d{2}-\d{2}-\d{4}$/.test(dateStr)) {
        const parts = dateStr.split("-");
        const date1 = new Date(`${parts[2]}-${parts[1]}-${parts[0]}`); // BUG-01 line 459
        const date2 = new Date(`${parts[2]}-${parts[0]}-${parts[1]}`); // BUG-01 line 460
        if (!isNaN(date1.getTime())) return date1;
        if (!isNaN(date2.getTime())) return date2;
      } else if (/^\d{8}$/.test(dateStr)) {
        const year = dateStr.substring(0, 4);
        const month = dateStr.substring(4, 6);
        const day = dateStr.substring(6, 8);
        const date = new Date(`${year}-${month}-${day}`);            // BUG-01 line 470
        if (!isNaN(date.getTime())) return date;
      }
    }
  }

  return null;
}
```
