```
/**
 * Compute the start-of-week date for the given date, given the user's
 * `firstDayOfWeek` setting (0 = Sunday ... 6 = Saturday).
 *
 * Uses the canonical `((day - firstDay + 7) % 7)` offset formula:
 *   - `(day - firstDay + 7)` shifts day so 0 means "is the week-start"
 *   - `% 7` wraps negative offsets back into [0..6]
 *   - Subtract from d.getDate() to walk back to the week-start
 *
 * Replaces the original formula `d.getDate() - day + (day < firstDay ? -7 : 0) + firstDay`
 * which produced incorrect results for [list specific failing cells from Task 1's
 * trace, e.g. "firstDay=6 (Saturday-start) when day=N"]. See BUG-05 trace at
 * .planning/phases/02-security-correctness/02-04-SUMMARY.md.
 */
private getStartOfWeek(date: Date): Date {
  const d = new Date(date);
  const day = d.getDay();
  const firstDay = this.plugin.settings.firstDayOfWeek;
  const offset = (day - firstDay + 7) % 7;
  d.setDate(d.getDate() - offset);
  return d;
}
```
